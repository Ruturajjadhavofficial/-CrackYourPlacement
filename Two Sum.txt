class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> mpp;
        for(int i=0;i<nums.size();i++){
            int curr=nums[i];
            int more_needed=target-curr;
            if(mpp.find(more_needed)!=mpp.end()){
                return {mpp[more_needed],i};
            }
            mpp[curr]=i;
        }
        return {-1,-1};
    }
};
